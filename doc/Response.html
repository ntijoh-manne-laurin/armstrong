def send
  @session.print "HTTP/1.1 #{@status}\r\n"
  @session.print "Content-length: #{@body.length}\r\n"
  @session.print "Content-Type: #{@content_type}\r\n"
  @session.print "\r\n"
  @session.print @body
  @session.close
end
